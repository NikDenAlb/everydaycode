package leetcode.random.task374;

/**
 * Forward declaration of guess API.
 *
 * @param num your guess
 * @return -1 if num is higher than the picked number
 * 1 if num is lower than the picked number
 * otherwise return 0
 * int guess(int num);
 */

class EasySolution extends GuessGame {
    public int guessNumber(int n) {
        int l = 1;
        while (true) {
            int m = l + (n - l) / 2;
            switch (guess(m)) {
                case -1 -> n = m - 1;
                case 1 -> l = m + 1;
                case 0 -> {
                    return m;
                }
            }
        }
    }
}
